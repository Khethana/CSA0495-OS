#include <stdio.h>
#include <pthread.h>

void *printMessage(void *arg) {
    printf("Hello from thread %d\n", *(int *)arg);
    return NULL;
}

int main() {
    pthread_t tid[2];
    int ids[2] = {1, 2};

    for (int i = 0; i < 2; i++)
        pthread_create(&tid[i], NULL, printMessage, &ids[i]);

    for (int i = 0; i < 2; i++)
        pthread_join(tid[i], NULL);

    return 0;
}
