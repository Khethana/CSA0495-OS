#include <stdio.h>

typedef struct {
    int id, bt, wt, tat;
} Process;

void sjfScheduling(Process p[], int n) {
    int i, j;
    Process temp;
    
    for (i = 0; i < n - 1; i++) 
        for (j = i + 1; j < n; j++) 
            if (p[i].bt > p[j].bt) { 
                temp = p[i]; 
                p[i] = p[j]; 
                p[j] = temp; 
            }
    
    p[0].wt = 0;
    for (i = 1; i < n; i++)
        p[i].wt = p[i-1].wt + p[i-1].bt;
    
    for (i = 0; i < n; i++)
        p[i].tat = p[i].wt + p[i].bt;
    
    for (i = 0; i < n; i++)
        printf("P%d: WT=%d, TAT=%d\n", p[i].id, p[i].wt, p[i].tat);
}

int main() {
    int n;
    printf("Enter number of processes: ");
    scanf("%d", &n);
    Process p[n];

    printf("Enter burst times: ");
    for (int i = 0; i < n; i++) {
        p[i].id = i + 1;
        scanf("%d", &p[i].bt);
    }

    sjfScheduling(p, n);
    return 0;
}
